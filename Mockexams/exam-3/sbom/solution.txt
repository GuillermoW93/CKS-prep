# install BOM
curl -L https://github.com/kubernetes-sigs/bom/releases/download/v0.6.0/bom_v0.6.0_linux_amd64 -o bom

# or raspberry pi ARM64
curl -L https://github.com/kubernetes-sigs/bom/releases/download/v0.6.0/bom-arm64-linux -o bom

# make it executable
chmod +x bom

# move to path
sudo mv /usr/local/bin/bom

# verify bom
bom version
